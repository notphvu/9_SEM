\begin{problem}[Expected value practice]
Let $X$ be a discrete random variable with $\Prob(X=0)=\Prob(X=1)=\tfrac12$.
Compute $\E[X]$ and $\Var(X)$.
\end{problem}

\begin{solution}
$\E[X]=\tfrac12$. Also $\Var(X)=\E[X^2]-\E[X]^2=\tfrac12-\tfrac14=\tfrac14$.
\end{solution}
